DELIMITER $$
use fus$$
DROP PROCEDURE IF EXISTS Create_New_Delivery$$
CREATE PROCEDURE Create_New_Delivery(	
		In new_PickupAddress	varchar(255)	, 
		IN new_DropDownAddress	varchar(255)	,
		IN new_CustomerID 		int				,
		IN new_CourierID 		int			
)
BEGIN
    INSERT INTO Delivery  (
				Date			,
        		PickupAddress	,
                DropDownAddress	,
                CustomerID 		,
                CourierID 		,
				IsActive
					
    )
    VALUES(
		sysdate()				,
        new_PickupAddress		, 
        new_DropDownAddress		,
        new_CustomerID 			,
        new_CourierID 			,
		0
        );
END $$
